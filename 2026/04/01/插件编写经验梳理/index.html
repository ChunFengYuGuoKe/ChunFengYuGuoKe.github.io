插件与协议包的概念不同，协议包用于设备主动连接的场景，插件则适用于

ka-mc-plugin

create table public.device_metrics (
name character varying(255), – 名称
type character varying(255), – 类型
count double precision, – 计数
value double precision, – 值
max double precision, – 最大值
total double precision, – 总计
active double precision, – 活跃数
unknown double precision, – 未知
sum double precision, – 合计
mean double precision, – 平均值
duration double precision, – 期间
“productId” character varying(255), – productId
id character varying(64) not null, – ID
timestamp timestamp without time zone not null – 时间戳
);
create unique index idx_device_metrics_id on device_metrics using btree (id, timestamp);
create index device_metrics_timestamp_idx on device_metrics using btree (timestamp);
comment on column public.device_metrics.name is ‘名称’;
comment on column public.device_metrics.type is ‘类型’;
comment on column public.device_metrics.count is ‘计数’;
comment on column public.device_metrics.value is ‘值’;
comment on column public.device_metrics.max is ‘最大值’;
comment on column public.device_metrics.total is ‘总计’;
comment on column public.device_metrics.active is ‘活跃数’;
comment on column public.device_metrics.unknown is ‘未知’;
comment on column public.device_metrics.sum is ‘合计’;
comment on column public.device_metrics.mean is ‘平均值’;
comment on column public.device_metrics.duration is ‘期间’;
comment on column public.device_metrics.”productId” is ‘productId’;
comment on column public.device_metrics.id is ‘ID’;
comment on column public.device_metrics.timestamp is ‘时间戳’;
